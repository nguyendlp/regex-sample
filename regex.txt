REGEX:

1. Integer: ^\s*[+-]?\d+\s*$
2. Slug: /^[a-z0-9-]+$/
3. Email: /^([a-z0-9_\.-]+)@([\da-z\.-]+)\.([a-z\.]{2,6})$/
4. URL: /^(https?:\/\/)?([\da-z\.-]+)\.([a-z\.]{2,6})([\/\w \.-]*)*\/?$/
5. IP Address:  /^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/
6. HTML tag: /^<([a-z]+)([^<]+)*(?:>(.*)<\/\1>|\s+\/>)$/
7. Hex value: /^#?([a-f0-9]{6}|[a-f0-9]{3})$/
8. Password: /^[a-z0-9_-]{6,18}$/
9  Username: /^[a-z0-9_-]{3,16}$/
